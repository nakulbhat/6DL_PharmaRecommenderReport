\begin{frame}{Key Takeaways}
  \begin{itemize}
    \item \textbf{FusionNet} best overall -- Jaccard 0.787, subset accuracy 0.646
    \pause
    \item \textbf{SelfAttention} best for top-$k$ retrieval -- highest PR-AUC \& Precision@$K$
    \pause
    \item All models ROC-AUC $> 0.993$ -- strong per-label discrimination
    \pause
    \item Beyond a representational threshold, \textbf{safety \& interpretability} become the key differentiators
  \end{itemize}
